210void initializeBasePst() {
211 for (Piece piece = WHITE_PAWN; piece <= BLACK_KING; piece++) {
212 for (Square square = A1; square <= H8; square++) {
213 const int midgamePieceValue = baseMaterialValues[MIDGAME][getPieceType(piece)];
214 const int endgamePieceValue = baseMaterialValues[ENDGAME][getPieceType(piece)];
215 const PieceColor color = getPieceColor(piece);
216
217 if (color == WHITE) {
218 baseMidgamePstTable[piece][square] = midgamePieceValue + getBaseMidgameTable(getPieceType(piece))[square ^ 56];
219 baseEndgamePstTable[piece][square] = endgamePieceValue + getBaseEndgameTable(getPieceType(piece))[square ^ 56];
220 } else {
221 baseMidgamePstTable[piece][square] = midgamePieceValue + getBaseMidgameTable(getPieceType(piece))[square];
222 baseEndgamePstTable[piece][square] = endgamePieceValue + getBaseEndgameTable(getPieceType(piece))[square];
223 }
224 }
225 }
226}
