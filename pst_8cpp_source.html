211void initializePst() {
212 for (Piece piece = WHITE_PAWN; piece <= BLACK_KING; piece++) {
213 for (Square square = A1; square <= H8; square++) {
214 const int midgamePieceValue = evalMaterialValues[MIDGAME][getPieceType(piece)];
215 const int endgamePieceValue = evalMaterialValues[ENDGAME][getPieceType(piece)];
216 const PieceColor color = getPieceColor(piece);
217
218 if (color == WHITE) {
219 midgamePstTable[piece][square] = midgamePieceValue + getMidgameTable(getPieceType(piece))[square ^ 56];
220 endgamePstTable[piece][square] = endgamePieceValue + getEndgameTable(getPieceType(piece))[square ^ 56];
221 } else {
222 midgamePstTable[piece][square] = midgamePieceValue + getMidgameTable(getPieceType(piece))[square];
223 endgamePstTable[piece][square] = endgamePieceValue + getEndgameTable(getPieceType(piece))[square];
224 }
225 }
226 }
