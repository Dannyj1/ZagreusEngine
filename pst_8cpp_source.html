201void initializePst() {
202 for (Piece piece = WHITE_PAWN; piece <= BLACK_KING; piece++) {
203 for (Square square = A1; square <= H8; square++) {
204 const int pieceValue = getPieceValue(piece);
205 const PieceColor color = getPieceColor(piece);
206
207 if (color == WHITE) {
208 midgamePst[piece][square] = pieceValue + getMidgameTable(getPieceType(piece))[square ^ 56];
209 endgamePst[piece][square] = pieceValue + getEndgameTable(getPieceType(piece))[square ^ 56];
210 } else {
211 midgamePst[piece][square] = pieceValue + getMidgameTable(getPieceType(piece))[square];
212 endgamePst[piece][square] = pieceValue + getEndgameTable(getPieceType(piece))[square];
213 }
214 }
215 }
216}
