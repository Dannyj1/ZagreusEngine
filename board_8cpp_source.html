35void initZobristConstants() {
36 pcg64_oneseq_once_insecure rng;
37
38 rng.seed(0x661778f67199663dULL);
39
40 for (uint64_t& zobrist : zobristConstants) {
41 zobrist = rng();
42
43 while (zobrist == 0) {
44 zobrist = rng();
45 }
46 }
47}

52constexpr uint64_t getZobristConstant(const int index) {
53 assert(index >= 0 && index < 781);
54 assert(zobristConstants[index] != 0);
55 return zobristConstants[index];

63template <PieceColor movedColor>
64bool Board::isPositionLegal() const {
65 constexpr PieceColor opponentColor = !movedColor;
66 constexpr Piece king = movedColor == WHITE ? WHITE_KING : BLACK_KING;
67 const uint64_t kingBB = getPieceBoard<king>();
68 const Square kingSquare = bitboardToSquare(kingBB);
69 const Move lastMove = getLastMove();
70 const MoveType lastMoveType = getMoveType(lastMove);
71
72 if (lastMoveType == CASTLING) {
73 const Square fromSquare = getFromSquare(lastMove);
74 const Square toSquare = getToSquare(lastMove);
75 const uint64_t fromSquareAttacks = getSquareAttackersByColor<opponentColor>(fromSquare);
76
77 if (fromSquareAttacks) {
78 // King was in check before castling
79 return false;
80 }
81
82 uint64_t castlingPath = 0;
83
84 if (toSquare == G1) {
85 castlingPath = WHITE_KINGSIDE_CASTLE_PATH;
86 } else if (toSquare == C1) {
87 castlingPath = WHITE_QUEENSIDE_CASTLE_PATH;
88 } else if (toSquare == G8) {
89 castlingPath = BLACK_KINGSIDE_CASTLE_PATH;
90 } else if (toSquare == C8) {
91 castlingPath = BLACK_QUEENSIDE_CASTLE_PATH;
92 }
93
94 while (castlingPath) {
95 const Square square = static_cast<Square>(popLsb(castlingPath));
96 const uint64_t attackers = getSquareAttackersByColor<opponentColor>(square);
97
98 if (attackers) {
99 return false;
100 }
101 }
102 }
103
104 return !getSquareAttackersByColor<opponentColor>(kingSquare);
105}

115template <CastlingRights side>
116bool Board::canCastle() const {
117 assert(side != WHITE_CASTLING && side != BLACK_CASTLING);
118
119 if (!(castlingRights & side)) {
120 return false;
121 }
122
123 uint64_t castlingPath = 0;
124
125 if constexpr (side == WHITE_KINGSIDE) {
126 castlingPath = WHITE_KINGSIDE_CASTLE_UNOCCUPIED;
127 } else if constexpr (side == WHITE_QUEENSIDE) {
128 castlingPath = WHITE_QUEENSIDE_CASTLE_UNOCCUPIED;
129 } else if constexpr (side == BLACK_KINGSIDE) {
130 castlingPath = BLACK_KINGSIDE_CASTLE_UNOCCUPIED;
131 } else if constexpr (side == BLACK_QUEENSIDE) {
132 castlingPath = BLACK_QUEENSIDE_CASTLE_UNOCCUPIED;
133 }
134
135 if (occupied & castlingPath) {
136 return false;
137 }
138
139 return true;
140}

152uint64_t Board::getSquareAttackers(const Square square) const {
153 assert(square < SQUARES);
154 const uint64_t knights = getPieceBoard<WHITE_KNIGHT>() | getPieceBoard<BLACK_KNIGHT>();
155 const uint64_t kings = getPieceBoard<WHITE_KING>() | getPieceBoard<BLACK_KING>();
156 uint64_t bishopsQueens = getPieceBoard<WHITE_QUEEN>() | getPieceBoard<BLACK_QUEEN>();
157 uint64_t rooksQueens = getPieceBoard<WHITE_QUEEN>() | getPieceBoard<BLACK_QUEEN>();
158 rooksQueens |= getPieceBoard<WHITE_ROOK>() | getPieceBoard<BLACK_ROOK>();
159 bishopsQueens |= getPieceBoard<WHITE_BISHOP>() | getPieceBoard<BLACK_BISHOP>();
160
161 return (getPawnAttacks<WHITE>(square) & getPieceBoard<BLACK_PAWN>())
162 | (getPawnAttacks<BLACK>(square) & getPieceBoard<WHITE_PAWN>())
163 | (getKnightAttacks(square) & knights)
164 | (getKingAttacks(square) & kings)
165 | (getBishopAttacks(square, occupied) & bishopsQueens)
166 | (getRookAttacks(square, occupied) & rooksQueens);
167}

211bool Board::isDraw() const {
212 const uint64_t occupied = getOccupiedBitboard();
213
214 if (halfMoveClock >= 100) {
215 return true;
216 }
217
218 const uint64_t zobristHash = getZobristHash();
219
220 // 3-fold repetition
221 for (int i = ply - 2; i >= 0; i -= 2) {
222 if (history[i].zobristHash == zobristHash) {
223 return true;
224 }
225 }
226
227 const uint64_t sufficientMaterial = getPieceBoard<WHITE_QUEEN>() | getPieceBoard<BLACK_QUEEN>()
228 | getPieceBoard<WHITE_ROOK>() | getPieceBoard<BLACK_ROOK>()
229 | getPieceBoard<WHITE_PAWN>() | getPieceBoard<BLACK_PAWN>();
230
231 if (sufficientMaterial) {
232 // Can never be a draw with queens, rooks or pawns
233 return false;
234 }
235
236 // Only kings left
237 if (popcnt(occupied) == 2) {
238 return true;
239 }
240
241 // Check for KBvK, KNvK
242 if (popcnt(occupied) == 3) {
243 if (getPieceBoard<WHITE_BISHOP>() || getPieceBoard<BLACK_BISHOP>() || getPieceBoard<WHITE_KNIGHT>() ||
244 getPieceBoard<
245 BLACK_KNIGHT>()) {
246 return true;
247 }
248 }
249
250 // Check for KBvKB where bishops are on the same color
251 if (popcnt(occupied) == 4) {
252 if (getPieceBoard<WHITE_BISHOP>() && getPieceBoard<BLACK_BISHOP>()) {
253 if ((getPieceBoard<WHITE_BISHOP>() & DARK_SQUARES) == (getPieceBoard<BLACK_BISHOP>() & DARK_SQUARES)) {
254 return true;
255 }
256
257 if ((getPieceBoard<WHITE_BISHOP>() & LIGHT_SQUARES) == (getPieceBoard<BLACK_BISHOP>() & LIGHT_SQUARES)) {
258 return true;
259 }
260 }
261 }
262
263 return false;
264}

277template <PieceColor color>
278Square Board::getSmallestAttacker(Square square) const {
279 const uint64_t attackers = getSquareAttackersByColor<color>(square);
280
281 if (!attackers) {
282 return NONE;
283 }
284
285 constexpr Piece attackerOrder[6] = {
286 color == WHITE ? WHITE_PAWN : BLACK_PAWN,
287 color == WHITE ? WHITE_KNIGHT : BLACK_KNIGHT,
288 color == WHITE ? WHITE_BISHOP : BLACK_BISHOP,
289 color == WHITE ? WHITE_ROOK : BLACK_ROOK,
290 color == WHITE ? WHITE_QUEEN : BLACK_QUEEN,
291 color == WHITE ? WHITE_KING : BLACK_KING
292 };
293
294 for (const Piece piece : attackerOrder) {
295 if (const uint64_t candidate = attackers & bitboards[piece]) {
296 return static_cast<Square>(bitscanForward(candidate));
297 }
298 }
299
300 return NONE;
301}

303int estimateMoveValue(const Board& board, const Move move) {
304 const MoveType moveType = getMoveType(move);
305 const Piece capturedPiece = board.getPieceOnSquare(getToSquare(move));
306 int value = getPieceValue(capturedPiece);
307
308 if (moveType == PROMOTION) {
309 value += getPieceValue(getPieceFromPromotionPiece(getPromotionPiece(move), board.getSideToMove())) -
310 getPieceValue(WHITE_PAWN);
311 } else if (moveType == EN_PASSANT) {
312 value = getPieceValue(WHITE_PAWN);
313 } else if (moveType == CASTLING) {
314 value = 0;
315 }
316
317 return value;
318}

326bool Board::see(const Move move, int threshold) {
327 const Square fromSquare = getFromSquare(move);
328 const Square toSquare = getToSquare(move);
329 const MoveType moveType = getMoveType(move);
330 const Piece nextVictim = moveType == PROMOTION
331 ? getPieceFromPromotionPiece(getPromotionPiece(move), sideToMove)
332 : getPieceOnSquare(fromSquare);
333 PieceType nextVictimType = getPieceType(nextVictim);
334 const int moveValue = estimateMoveValue(*this, move);
335 int balance = moveValue - threshold;
336
337 if (balance < 0) {
338 return false;
339 }
340
341 balance -= getPieceValue(nextVictim);
342
343 if (balance >= 0) {
344 return true;
345 }
346
347 const uint64_t bishops = getPieceBoard<WHITE_BISHOP>() | getPieceBoard<BLACK_BISHOP>() | getPieceBoard<
348 WHITE_QUEEN>() | getPieceBoard<BLACK_QUEEN>();
349 const uint64_t rooks = getPieceBoard<WHITE_ROOK>() | getPieceBoard<BLACK_ROOK>() | getPieceBoard<
350 WHITE_QUEEN>() | getPieceBoard<BLACK_QUEEN>();
351 const uint64_t oldOccupied = getOccupiedBitboard();
352 const PieceColor oldSideToMove = sideToMove;
353 occupied = (occupied ^ squareToBitboard(fromSquare)) | squareToBitboard(toSquare);
354
355 if (moveType == EN_PASSANT) {
356 occupied ^= squareToBitboard(enPassantSquare);
357 }
358
359 uint64_t attackers = getSquareAttackers(toSquare) & occupied;
360 sideToMove = !sideToMove;
361
362 while (true) {
363 const uint64_t ownAttackers = attackers & getColorBitboard(sideToMove);
364
365 if (!ownAttackers) {
366 break;
367 }
368
369 for (nextVictimType = PAWN; nextVictimType <= QUEEN;
370 nextVictimType = static_cast<PieceType>(nextVictimType + 1)) {
371 if (ownAttackers & getPieceBoard(getPieceFromType(nextVictimType, sideToMove))) {
372 break;
373 }
374 }
375
376 occupied ^= squareToBitboard(
377 bitscanForward(ownAttackers & getPieceBoard(getPieceFromType(nextVictimType, sideToMove))));
378
379 if (nextVictimType == PAWN || nextVictimType == BISHOP || nextVictimType == QUEEN) {
380 attackers |= getBishopAttacks(toSquare, occupied) & bishops;
381 }
382
383 if (nextVictimType == ROOK || nextVictimType == QUEEN) {
384 attackers |= getRookAttacks(toSquare, occupied) & rooks;
385 }
386
387 attackers &= occupied;
388 balance = -balance - 1 - getPieceValue(getPieceFromType(nextVictimType, sideToMove));
389 sideToMove = !sideToMove;
390
391 if (balance >= 0) {
392 if (nextVictimType == KING && (attackers & getColorBitboard(sideToMove))) {
393 sideToMove = !sideToMove;
394 }
395
396 break;
397 }
398 }
399
400 bool result = sideToMove != oldSideToMove;
401 occupied = oldOccupied;
402 sideToMove = oldSideToMove;
403 return result;
404}

410void Board::makeMove(const Move move) {
411 const uint8_t fromSquare = getFromSquare(move);
412 const uint8_t toSquare = getToSquare(move);
413 const MoveType moveType = getMoveType(move);
414 const Piece movedPiece = getPieceOnSquare(fromSquare);
415 const PieceType movedPieceType = getPieceType(movedPiece);
416 const Piece capturedPiece = getPieceOnSquare(toSquare);
417
418 history[ply].move = move;
419 history[ply].previousMove = previousMove;
420 history[ply].capturedPiece = capturedPiece;
421 history[ply].enPassantSquare = enPassantSquare;
422
423 if (enPassantSquare != 255) {
424 zobristHash ^= getZobristConstant(ZOBRIST_EN_PASSANT_START_INDEX + (enPassantSquare % 8));
425 }
426
427 enPassantSquare = 255;
428 history[ply].castlingRights = castlingRights;
429 history[ply].zobristHash = zobristHash;
430 history[ply].halfMoveClock = halfMoveClock;
431
432 halfMoveClock += 1;
433
434 if (capturedPiece != EMPTY) {
435 removePiece(capturedPiece, toSquare);
436 halfMoveClock = 0;
437
438 if (capturedPiece == WHITE_ROOK) {
439 if (toSquare == A1) {
440 if (castlingRights & WHITE_QUEENSIDE) {
441 castlingRights &= ~WHITE_QUEENSIDE;
442 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_QUEENSIDE_INDEX);
443 }
444 } else if (toSquare == H1) {
445 if (castlingRights & WHITE_KINGSIDE) {
446 castlingRights &= ~WHITE_KINGSIDE;
447 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_KINGSIDE_INDEX);
448 }
449 }
450 } else if (capturedPiece == BLACK_ROOK) {
451 if (toSquare == A8) {
452 if (castlingRights & BLACK_QUEENSIDE) {
453 castlingRights &= ~BLACK_QUEENSIDE;
454 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_QUEENSIDE_INDEX);
455 }
456 } else if (toSquare == H8) {
457 if (castlingRights & BLACK_KINGSIDE) {
458 castlingRights &= ~BLACK_KINGSIDE;
459 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_KINGSIDE_INDEX);
460 }
461 }
462 }
463 }
464
465 removePiece(movedPiece, fromSquare);
466
467 if (moveType == PROMOTION) {
468 const PieceColor color = getPieceColor(movedPiece);
469 const PromotionPiece promotionPieceType = getPromotionPiece(move);
470 const Piece promotionPiece = getPieceFromPromotionPiece(promotionPieceType, color);
471
472 setPiece(promotionPiece, toSquare);
473 } else {
474 setPiece(movedPiece, toSquare);
475 }
476
477 if (moveType == EN_PASSANT) {
478 if (sideToMove == WHITE) {
479 removePiece(BLACK_PAWN, toSquare + SOUTH);
480 } else {
481 removePiece(WHITE_PAWN, toSquare + NORTH);
482 }
483 } else if (moveType == CASTLING) {
484 if (toSquare == G1) {
485 removePiece(WHITE_ROOK, H1);
486 setPiece(WHITE_ROOK, F1);
487 } else if (toSquare == C1) {
488 removePiece(WHITE_ROOK, A1);
489 setPiece(WHITE_ROOK, D1);
490 } else if (toSquare == G8) {
491 removePiece(BLACK_ROOK, H8);
492 setPiece(BLACK_ROOK, F8);
493 } else if (toSquare == C8) {
494 removePiece(BLACK_ROOK, A8);
495 setPiece(BLACK_ROOK, D8);
496 }
497
498 if (sideToMove == WHITE) {
499 if (castlingRights & WHITE_KINGSIDE) {
500 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_KINGSIDE_INDEX);
501 }
502
503 if (castlingRights & WHITE_QUEENSIDE) {
504 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_QUEENSIDE_INDEX);
505 }
506
507 castlingRights &= ~WHITE_CASTLING;
508 } else {
509 if (castlingRights & BLACK_KINGSIDE) {
510 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_KINGSIDE_INDEX);
511 }
512
513 if (castlingRights & BLACK_QUEENSIDE) {
514 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_QUEENSIDE_INDEX);
515 }
516
517 castlingRights &= ~BLACK_CASTLING;
518 }
519 }
520
521 if (movedPiece == WHITE_KING) {
522 if (castlingRights & WHITE_KINGSIDE) {
523 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_KINGSIDE_INDEX);
524 }
525
526 if (castlingRights & WHITE_QUEENSIDE) {
527 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_QUEENSIDE_INDEX);
528 }
529
530 castlingRights &= ~WHITE_CASTLING;
531 } else if (movedPiece == BLACK_KING) {
532 if (castlingRights & BLACK_KINGSIDE) {
533 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_KINGSIDE_INDEX);
534 }
535
536 if (castlingRights & BLACK_QUEENSIDE) {
537 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_QUEENSIDE_INDEX);
538 }
539
540 castlingRights &= ~BLACK_CASTLING;
541 } else if (movedPiece == WHITE_ROOK) {
542 if (fromSquare == A1) {
543 if (castlingRights & WHITE_QUEENSIDE) {
544 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_QUEENSIDE_INDEX);
545 castlingRights &= ~WHITE_QUEENSIDE;
546 }
547 } else if (fromSquare == H1) {
548 if (castlingRights & WHITE_KINGSIDE) {
549 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_KINGSIDE_INDEX);
550 castlingRights &= ~WHITE_KINGSIDE;
551 }
552 }
553 } else if (movedPiece == BLACK_ROOK) {
554 if (fromSquare == A8) {
555 if (castlingRights & BLACK_QUEENSIDE) {
556 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_QUEENSIDE_INDEX);
557 castlingRights &= ~BLACK_QUEENSIDE;
558 }
559 } else if (fromSquare == H8) {
560 if (castlingRights & BLACK_KINGSIDE) {
561 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_KINGSIDE_INDEX);
562 castlingRights &= ~BLACK_KINGSIDE;
563 }
564 }
565 }
566
567 if (movedPieceType == PAWN) {
568 halfMoveClock = 0;
569
570 if ((fromSquare ^ toSquare) == 16) {
571 if (sideToMove == WHITE) {
572 enPassantSquare = toSquare + SOUTH;
573 zobristHash ^= getZobristConstant(ZOBRIST_EN_PASSANT_START_INDEX + (enPassantSquare % 8));
574 } else {
575 enPassantSquare = toSquare + NORTH;
576 zobristHash ^= getZobristConstant(ZOBRIST_EN_PASSANT_START_INDEX + (enPassantSquare % 8));
577 }
578 }
579 }
580
581 if (sideToMove == BLACK) {
582 fullmoveClock += 1;
583 }
584
585 previousMove = move;
586 sideToMove = !sideToMove;
587 zobristHash ^= getZobristConstant(ZOBRIST_SIDE_TO_MOVE_INDEX);
588
589 assert(ply < MAX_PLY);
590 ply++;
591 assert(ply < MAX_PLY);
592 assert(enPassantSquare == 255 || (enPassantSquare / 8 == 2 || enPassantSquare / 8 == 5));
593}

598void Board::unmakeMove() {
599 ply--;
600 assert(ply >= 0 && ply < MAX_PLY);
601 const BoardState& state = history[ply];
602 const uint8_t fromSquare = getFromSquare(state.move);
603 const uint8_t toSquare = getToSquare(state.move);
604 const MoveType moveType = getMoveType(state.move);
605 Piece movedPiece = getPieceOnSquare(toSquare);
606 const PieceColor movedColor = getPieceColor(movedPiece);
607
608 removePiece(movedPiece, toSquare);
609
610 if (moveType == PROMOTION) {
611 const PieceColor color = getPieceColor(movedPiece);
612 movedPiece = static_cast<Piece>(WHITE_PAWN + color);
613 }
614
615 setPiece(movedPiece, fromSquare);
616
617 if (state.capturedPiece != EMPTY) {
618 setPiece(state.capturedPiece, toSquare);
619 }
620
621 if (moveType == EN_PASSANT) {
622 const PieceColor movedPieceColor = getPieceColor(movedPiece);
623
624 if (movedPieceColor == WHITE) {
625 setPiece(BLACK_PAWN, toSquare + SOUTH);
626 } else {
627 setPiece(WHITE_PAWN, toSquare + NORTH);
628 }
629 }
630
631 if (moveType == CASTLING) {
632 if (toSquare == G1) {
633 removePiece(WHITE_ROOK, F1);
634 setPiece(WHITE_ROOK, H1);
635 } else if (toSquare == C1) {
636 removePiece(WHITE_ROOK, D1);
637 setPiece(WHITE_ROOK, A1);
638 } else if (toSquare == G8) {
639 removePiece(BLACK_ROOK, F8);
640 setPiece(BLACK_ROOK, H8);
641 } else if (toSquare == C8) {
642 removePiece(BLACK_ROOK, D8);
643 setPiece(BLACK_ROOK, A8);
644 }
645 }
646
647 if (movedColor == BLACK) {
648 fullmoveClock -= 1;
649 }
650
651 assert(state.move != NO_MOVE);
652 this->previousMove = state.previousMove;
653 this->halfMoveClock = state.halfMoveClock;
654 this->sideToMove = !sideToMove;
655 this->enPassantSquare = state.enPassantSquare;
656 this->castlingRights = state.castlingRights;
657 this->zobristHash = state.zobristHash;
658}

660void Board::makeNullMove() {
661 history[ply].move = NO_MOVE;
662 history[ply].previousMove = previousMove;
663 history[ply].capturedPiece = EMPTY;
664 history[ply].enPassantSquare = enPassantSquare;
665
666 if (enPassantSquare != 255) {
667 zobristHash ^= getZobristConstant(ZOBRIST_EN_PASSANT_START_INDEX + (enPassantSquare % 8));
668 }
669
670 enPassantSquare = 255;
671 history[ply].castlingRights = castlingRights;
672 history[ply].zobristHash = zobristHash;
673 history[ply].halfMoveClock = halfMoveClock;
674
675 halfMoveClock += 1;
676
677 if (sideToMove == BLACK) {
678 fullmoveClock += 1;
679 }
680
681 previousMove = NO_MOVE;
682 sideToMove = !sideToMove;
683 zobristHash ^= getZobristConstant(ZOBRIST_SIDE_TO_MOVE_INDEX);
684
685 assert(ply < MAX_PLY);
686 ply++;
687 assert(ply < MAX_PLY);
688}

690void Board::unmakeNullMove() {
691 ply--;
692 assert(ply >= 0 && ply < MAX_PLY);
693 const BoardState& state = history[ply];
694
695 if (sideToMove == BLACK) {
696 fullmoveClock -= 1;
697 }
698
699 assert(state.move == NO_MOVE);
700 this->previousMove = state.previousMove;

763bool Board::setFromFEN(const std::string_view fen) {
764 uint8_t index = A8;
765 int spaces = 0;
766
767 reset();
768
769 for (const char character : fen) {
770 if (character == ' ') {
771 spaces++;
772 continue;
773 }
774
775 if (character == ',') {
776 break;
777 }
778
779 if (spaces == 0) {
780 if (character == '/') {
781 index -= 16;
782 continue;
783 }
784
785 if (character >= '1' && character <= '8') {
786 index += character - '0';
787 continue;
788 }
789
790 if (character >= 'A' && character <= 'z') {
791 setPieceFromFENChar(character, index);
792 index++;
793 } else {
794 return false;
795 }
796 }
797
798 if (spaces == 1) {
799 if (tolower(character) == 'w') {
800 sideToMove = WHITE;
801 } else if (tolower(character) == 'b') {
802 sideToMove = BLACK;
803 zobristHash ^= getZobristConstant(ZOBRIST_SIDE_TO_MOVE_INDEX);
804 } else {
805 return false;
806 }
807 }
808
809 if (spaces == 2) {
810 if (character == '-') {
811 continue;
812 } else if (character == 'K') {
813 castlingRights |= WHITE_KINGSIDE;
814 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_KINGSIDE_INDEX);
815 continue;
816 }
817
818 if (character == 'Q') {
819 castlingRights |= WHITE_QUEENSIDE;
820 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_QUEENSIDE_INDEX);
821 continue;
822 }
823
824 if (character == 'k') {
825 castlingRights |= BLACK_KINGSIDE;
826 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_KINGSIDE_INDEX);
827 continue;
828 }
829
830 if (character == 'q') {
831 castlingRights |= BLACK_QUEENSIDE;
832 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_QUEENSIDE_INDEX);
833 continue;
834 }
835
836 return false;
837 }
838
839 if (spaces == 3) {
840 if (character == '-') {
841 continue;
842 }
843
844 if (tolower(character) < 'a' || tolower(character) > 'h') {
845 continue;
846 }
847
848 const int8_t file = tolower(character) - 'a';
849 const int8_t rank = (!sideToMove) == WHITE ? 2 : 5;
850
851 if (file < 0 || file > 7) {
852 return false;
853 }
854
855 enPassantSquare = rank * 8 + file;
856 zobristHash ^= getZobristConstant(ZOBRIST_EN_PASSANT_START_INDEX + (enPassantSquare % 8));
857 index += 2;
858 }
859
860 if (spaces == 4) {
861 if (character >= '0' && character <= '9') {
862 halfMoveClock = halfMoveClock * 10 + (character - '0');
863 }
864 }
865
866 if (spaces == 5) {
867 if (character >= '0' && character <= '9') {
868 fullmoveClock = fullmoveClock * 10 + (character - '0');
869 }
870 }
871 }
872
873 return true;
874}
