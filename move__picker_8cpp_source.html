35bool MovePicker::next(Move& move) {
36 if (currentIndex >= moveList.size) {
37 return false;
38 }
39
40 int bestIndex = currentIndex;
41 int bestScore = scores[currentIndex];
42
43 for (int i = currentIndex + 1; i < moveList.size; ++i) {
44 if (scores[i] > bestScore) {
45 bestIndex = i;
46 bestScore = scores[i];
47 }
48 }
49
50 std::swap(scores[currentIndex], scores[bestIndex]);
51 std::swap(moveList.moves[currentIndex], moveList.moves[bestIndex]);
52
53 move = moveList.moves[currentIndex];
54
55 if (move == NO_MOVE) {
56 return false;
57 }
58
59 currentIndex += 1;
60 return true;
61}

72void MovePicker::score(Board& board) {
73 // TODO: Make better system that supports assigns a score to moves and sorts based on those
74 const PvLine& pvLine = board.getPreviousPvLine();
75 const int currentPly = board.getPly();
76 const int pvMoveIndex = currentPly - pvLine.startPly;
77 Move pvMove = NO_MOVE;
78 Move ttMove = NO_MOVE;
79
80 if (pvLine.moveCount >= pvMoveIndex) {
81 pvMove = pvLine.moves[pvMoveIndex];
82 }
83
84 uint64_t zobristHash = board.getZobristHash();
85 TTEntry* entry = tt->getEntry(zobristHash);
86
87 if (entry != nullptr) {
88 ttMove = entry->bestMove;
89
90 if (ttMove == pvMove) {
91 ttMove = NO_MOVE;
92 }
93 }
94
95 for (int i = 0; i < moveList.size; ++i) {
96 const Move move = moveList.moves[i];
97 const Square toSquare = getToSquare(move);
98 const Piece capturedPiece = board.getPieceOnSquare(toSquare);
99
100 if (move == pvMove) {
101 scores[i] = 5000000;
102 } else if (move == ttMove) {
103 scores[i] = 2500000;
104 } else if (capturedPiece != EMPTY) {
105 // MVV-LVA
106 const PieceType movingPiece = getPieceType(board.getPieceOnSquare(getFromSquare(move)));
107 // 200 * Piece Value - Index of the captured piece. Factor of 200 so it's always above history scores. Should be changed once we have staged movegen.
108 const int mvvLva = 200 * getPieceValue(capturedPiece) - static_cast<int>(movingPiece);
109
110 scores[i] = mvvLva;
111 } else {
112 // Ordering of quiet moves
113 const int historyValue = tt->getHistoryValue(board.getSideToMove(), move);
114
115 scores[i] = historyValue;
116 }
117 }
118}
