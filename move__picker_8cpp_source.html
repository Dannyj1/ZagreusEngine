59void MovePicker::sort(Board& board) {
60 // TODO: Make better system that supports assigns a score to moves and sorts based on those
61 const PvLine& pvLine = board.getPreviousPvLine();
62 const int currentPly = board.getPly();
63 const int pvMoveIndex = currentPly - pvLine.startPly;
64 Move pvMove = NO_MOVE;
65 Move ttMove = NO_MOVE;
66
67 if (pvLine.moveCount >= pvMoveIndex) {
68 pvMove = pvLine.moves[pvMoveIndex];
69 }
70
71 uint64_t zobristHash = board.getZobristHash();
72 TTEntry* entry = tt->getEntry(zobristHash);
73
74 if (entry != nullptr) {
75 ttMove = entry->bestMove;
76
77 if (ttMove == pvMove) {
78 ttMove = NO_MOVE;
79 }
80 }
81
82 for (int i = 0; i < moveList.size; ++i) {
83 const Move move = moveList.moves[i];
84 const Square toSquare = getToSquare(move);
85 const Square fromSquare = getFromSquare(move);

87
88 if (move == pvMove) {
89 scores[i] = 5000000;
90 } else if (move == ttMove) {
91 scores[i] = 2500000;
92 } else if (capturedPiece != EMPTY) {
93 // MVV-LVA
94 const PieceType movingPiece = getPieceType(board.getPieceOnSquare(getFromSquare(move)));
95 // 200 * Piece Value - Index of the captured piece. Factor of 200 so it's always above history scores. Should be changed once we have staged movegen.
96 const int mvvLva = 200 * getPieceValue(capturedPiece) - static_cast<int>(movingPiece);
97
98 scores[i] = mvvLva;
99 } else {
100 const Move previousMove = board.getPreviousMove();
101 // const Move counterMove = tt->counterMoves[getFromSquare(previousMove)][getToSquare(previousMove)];
102
103 // if (move == counterMove) {
104 // scores[i] = 50000;
105 // } else {
106 // Ordering of quiet moves
107 const int historyValue = tt->getHistoryValue(board.getSideToMove(), move);
108
109 scores[i] = historyValue;
110 // }
111 }
112 }
113
114 // Sort the moves after scoring them
115 for (int i = 0; i < moveList.size; ++i) {
116 for (int j = i + 1; j < moveList.size; ++j) {
117 if (scores[j] > scores[i]) {
