38void Evaluation::addScore(const int midgameScore, const int endgameScore) {
39 if (color == WHITE) {
40 whiteMidgameScore += midgameScore;
41 whiteEndgameScore += endgameScore;
42 } else {
43 blackMidgameScore += midgameScore;
44 blackEndgameScore += endgameScore;
45 }

57int Evaluation::evaluate() {
58 const int phase = calculatePhase();
59 const int modifier = board.getSideToMove() == WHITE ? 1 : -1;
60
61 initializeEvalData();
62
63 evaluatePieces();
64
65 const int whiteScore = ((whiteMidgameScore * (256 - phase)) + (whiteEndgameScore * phase)) / 256;
66 const int blackScore = ((blackMidgameScore * (256 - phase)) + (blackEndgameScore * phase)) / 256;
67
68 return (whiteScore - blackScore) * modifier;
69}
70
71constexpr int knightPhase = 1;
72constexpr int bishopPhase = 1;
73constexpr int rookPhase = 2;
74constexpr int queenPhase = 4;
75constexpr int totalPhase = knightPhase * 4 + bishopPhase * 4 + rookPhase * 4 + queenPhase * 2;
76
86int Evaluation::calculatePhase() const {
87 int phase = totalPhase;
88
89 phase -= popcnt(board.getPieceBoard<WHITE_KNIGHT>() | board.getPieceBoard<BLACK_KNIGHT>()) * knightPhase;
90 phase -= popcnt(board.getPieceBoard<WHITE_BISHOP>() | board.getPieceBoard<BLACK_BISHOP>()) * bishopPhase;
91 phase -= popcnt(board.getPieceBoard<WHITE_ROOK>() | board.getPieceBoard<BLACK_ROOK>()) * rookPhase;
92 phase -= popcnt(board.getPieceBoard<WHITE_QUEEN>() | board.getPieceBoard<BLACK_QUEEN>()) * queenPhase;
93
94 return (phase * 256 + (totalPhase / 2)) / totalPhase;
95}

97void Evaluation::evaluatePieces() {
98 evaluatePawns<WHITE>();
99 evaluatePawns<BLACK>();
100
101 // Exclude pawn attacks from mobility after evalutePawns filled the attack tables
102 evalData.mobilityArea[WHITE] &= ~evalData.attacksByPiece[WHITE_PAWN];
103 evalData.mobilityArea[BLACK] &= ~evalData.attacksByPiece[BLACK_PAWN];
104
105 evaluateKnights<WHITE>();
106 evaluateKnights<BLACK>();
107
108 evaluateBishops<WHITE>();
109 evaluateBishops<BLACK>();
110
111 evaluateRooks<WHITE>();
112 evaluateRooks<BLACK>();
113
114 evaluateQueens<WHITE>();
115 evaluateQueens<BLACK>();
116
117 evaluateKing<WHITE>();
118 evaluateKing<BLACK>();
119}

121template <PieceColor color>
122void Evaluation::evaluatePawns() {
123 constexpr Piece pawnPiece = color == WHITE ? WHITE_PAWN : BLACK_PAWN;
124 uint64_t pawns = board.getPieceBoard<pawnPiece>();
125
126 while (pawns) {
127 const Square square = static_cast<Square>(popLsb(pawns));
128 const int midgamePst = midgamePstTable[pawnPiece][square];
129 const int endgamePst = endgamePstTable[pawnPiece][square];
130
131#ifdef ZAGREUS_TUNER
132 trace.material[color][PAWN] += 1;
133 trace.pst[color][PAWN][square] += 1;
134#endif
135
136 addScore<color>(midgamePst, endgamePst);
137
138 const uint64_t attacks = getPawnAttacks<color>(square);
139
140 evalData.attacksFrom[square] = attacks;
141 evalData.attackedBy2[color] |= (attacks & evalData.attacksByColor[color]);
142 evalData.attacksByColor[color] |= attacks;
143 evalData.attacksByPiece[pawnPiece] |= attacks;
144 }
145}

335int getPieceValue(const Piece piece) {
336 switch (piece) {
337 case WHITE_PAWN:
338 case BLACK_PAWN:

340 case WHITE_KNIGHT:
341 case BLACK_KNIGHT:

346 case WHITE_ROOK:
