126void Engine::handleSetOptionCommand(const std::string& args) {
127 std::istringstream iss(args);
128 std::string arg;
129 std::string section;
130 std::string name;
131 std::string value;
132
133 while (iss >> arg) {
134 std::string lowercaseWord = arg;
135 std::ranges::transform(lowercaseWord, lowercaseWord.begin(),
136 [](const unsigned char c) { return std::tolower(c); });
137
138 if (lowercaseWord == "name") {
139 section = arg;
140 continue;
141 }
142
143 if (lowercaseWord == "value") {
144 section = arg;
145 continue;
146 }
147
148 if (section == "name") {
149 if (name.empty()) {
150 name = arg;
151 } else {
152 name += " " + arg;
153 }
154 }
155
156 if (section == "value") {
157 if (value.empty()) {
158 value = arg;
159 } else {
160 value += " " + arg;
161 }
162 }
163 }
164
165 if (name.empty()) {
166 sendMessage("ERROR: No option name provided.");
167 return;
168 }
169
170 if (!hasOption(name)) {
171 sendMessage("ERROR: Option " + name + " does not exist.");
172 return;
173 }
174
175 UCIOption& option = this->getOption(name);
176
177 if (value.empty()) {
178 if (option.getOptionType() == Button) {
179 if (option.getValue() == "true") {
180 value = "false";
181 } else {
182 value = "true";
183 }
184 } else {
185 sendMessage("ERROR: No option value provided.");
186 return;
187 }
188 }
189
190 option.setValue(value);
191
192 if (!didSetup) {
193 doSetup();
194 } else if (name == "Hash") {
195 TranspositionTable::getTT()->setTableSize(std::stoi(value));
196 }
197}

204void Engine::handlePositionCommand(const std::string_view args) {
205 // If the first arg is not "startpos" or "fen", report invalid usage
206 if (!args.starts_with("startpos") && !args.starts_with("fen")) {
207 sendMessage("ERROR: Invalid usage of position command.");
208 return;
209 }
210
211 // Go through each argument one by one
212 std::istringstream iss((args.data()));
213 std::string arg;
214
215 // If the first argument is "startpos", set the board to the starting position
216 iss >> arg;
217
218 if (arg == "startpos") {
219 board.setFromFEN(startPosFEN);
220 } else if (arg == "fen") {
221 std::string fen;
222
223 // Append args to the fen string until we reach the "moves" keyword or the end of the string
224 while (iss >> arg && arg != "moves") {
225 if (fen.empty()) {
226 fen = arg;
227 } else {
228 fen += " " + arg;
229 }
230 }
231
232 if (fen.empty()) {
233 sendMessage("ERROR: No FEN string provided.");
234 return;
235 }
236
237 if (!board.setFromFEN(fen)) {
238 sendMessage("ERROR: Invalid FEN string provided.");
239 return;
240 }
241 }
242
243 while (iss >> arg) {
244 if (arg == "moves") {
245 continue;
246 }
247
248 if (arg.size() != 4 && arg.size() != 5) {
249 sendMessage("ERROR: Invalid move notation provided.");
250 return;
251 }
252
253 Move move = getMoveFromMoveNotation(arg);
254
255 // TODO: Move this to the getMoveFromMoveNotation function, need to work around circular dependency first
256 Piece movedPiece = board.getPieceOnSquare(getFromSquare(move));
257
258 if (movedPiece == WHITE_KING) {
259 if (getFromSquare(move) == E1 && getToSquare(move) == G1 && board.canCastle<WHITE_KINGSIDE>()) {
260 move = encodeMove(E1, G1, CASTLING);
261 } else if (getFromSquare(move) == E1 && getToSquare(move) == C1 && board.canCastle<WHITE_QUEENSIDE>()) {
262 move = encodeMove(E1, C1, CASTLING);
263 }
264 } else if (movedPiece == BLACK_KING) {
265 if (getFromSquare(move) == E8 && getToSquare(move) == G8 && board.canCastle<BLACK_KINGSIDE>()) {
266 move = encodeMove(E8, G8, CASTLING);
267 } else if (getFromSquare(move) == E8 && getToSquare(move) == C8 && board.canCastle<BLACK_QUEENSIDE>()) {
268 move = encodeMove(E8, C8, CASTLING);
269 }
270 } else if (movedPiece == WHITE_PAWN || movedPiece == BLACK_PAWN) {
271 if (getToSquare(move) == board.getEnPassantSquare()) {
272 move = encodeMove(getFromSquare(move), getToSquare(move), EN_PASSANT);
273 }
274 }
275
276 board.makeMove(move);
277 }

280void Engine::handleGoCommand(std::string_view args) {
281 if (!didSetup) {
282 doSetup();
283 }
284
285 // If the board is empty, set it to the starting position
286 if (board.getOccupiedBitboard() == 0ULL) {
287 if (!board.setFromFEN(startPosFEN)) {
288 sendMessage("ERROR: Could not initialize default position.");
289 return;
290 }
291 }
292
293 std::istringstream iss(args.data());
294 std::string arg;
295
296 // Parse arguments. Currently, only wtime and btime or supported, just ignore the others.
297 uint32_t whiteTime = 0;
298 uint32_t blackTime = 0;
299 uint32_t whiteInc = 0;
300 uint32_t blackInc = 0;
301 uint16_t depth = 0;
302
303 while (iss >> arg) {
304 if (arg == "wtime") {
305 iss >> whiteTime;
306 } else if (arg == "btime") {
307 iss >> blackTime;
308 } else if (arg == "depth") {
309 iss >> depth;
310 } else if (arg == "winc") {
311 iss >> whiteInc;
312 } else if (arg == "binc") {
313 iss >> blackInc;
314 }
315 }
316
317 if (whiteTime == 0 && blackTime == 0 && depth == 0) {
318 // TODO: implement support for other args
319 sendMessage("ERROR: No time control or depth limit provided.");
320 return;
321 }
322
323 this->searchStopped = false;
324 SearchParams params{};
325 SearchStats stats{};
326
327 params.whiteTime = whiteTime;
328 params.blackTime = blackTime;
329 params.whiteInc = whiteInc;
330 params.blackInc = blackInc;
331 params.depth = depth;
332
333 Move bestMove;
334
335 if (board.getSideToMove() == WHITE) {
336 bestMove = search<WHITE>(*this, board, params, stats);
337 } else {
338 bestMove = search<BLACK>(*this, board, params, stats);
339 }
340
341 sendMessage("bestmove " + getMoveNotation(bestMove));
342}

355void Engine::handlePerftCommand(const std::string& args) {
356 if (!didSetup) {
357 doSetup();
358 }
359
360 if (args.empty() || args == " " || args == "\n") {
361 sendMessage("ERROR: No depth provided.");
362 return;
363 }
364
365 if (args.contains(' ')) {
366 sendMessage("ERROR: Too many arguments provided.");
367 return;
368 }
369
370 int depth = 0;
371
372 try {
373 depth = std::stoi(args);
374 } catch (const std::invalid_argument& e) {
375 sendMessage("ERROR: Depth must be an integer.");
376 return;
377 }
378
379 if (depth <= 0) {
380 sendMessage("ERROR: Depth must be at least 1.");
381 return;
382 }
383
384 if (board.getOccupiedBitboard() == 0ULL) {
385 board.setFromFEN(startPosFEN);
386 }
387
388 const auto start = std::chrono::high_resolution_clock::now();
389 const uint64_t nodes = perft(board, depth);
390 const auto end = std::chrono::high_resolution_clock::now();
391 const std::string tookSeconds = std::to_string(std::chrono::duration<double>(end - start).count());
392
393 sendInfoMessage(
394 "Depth: " + std::to_string(depth) + ", Nodes: " + std::to_string(nodes) + ", Time: " + tookSeconds + "s");
395}

401void Engine::processCommand(const std::string_view command, const std::string& args) {
402 if (command == "uci") {
403 handleUciCommand();
404 } else if (command == "debug") {
405 handleDebugCommand(args);
406 } else if (command == "isready") {
407 handleIsReadyCommand(args);
408 } else if (command == "setoption") {
409 handleSetOptionCommand(args);
410 } else if (command == "register") {
411 // Ignore this command, not relevant to this engine
412 } else if (command == "ucinewgame") {
413 handleUciNewGameCommand();
414 } else if (command == "position") {
415 handlePositionCommand(args);
416 } else if (command == "go") {
417 std::thread searchThread{&Engine::handleGoCommand, this, args};
418
419 searchThread.detach();
420 } else if (command == "stop") {
421 handleStopCommand();
422 } else if (command == "ponderhit") {
423 handlePonderHitCommand(args);
424 } else if (command == "quit") {
425 handleQuitCommand(args);
426 } else if (command == "perft") {
427 handlePerftCommand(args);
428 } else if (command == "print") {
429 handlePrintCommand();
430 } else {
431 // If unknown, we must skip it and process the rest.
432 if (args.empty() || args == " " || args == "\n") {
433 std::cout << "Unknown command: " << command << std::endl;
434 return;
435 }
436
437 std::string newCommand;
438 std::string newArgs;
439
440 if (args.contains(' ')) {
441 newCommand = args.substr(0, args.find(' '));
442 newArgs = args.substr(args.find(' ') + 1);
443 } else {
444 newCommand = args;
445 }
446
447 processCommand(newCommand, newArgs);
448 }
449}

520std::string removeRedundantSpaces(const std::string_view input) {
521 std::string result;
522 bool inSpace = false; // Track if we are in a sequence of spaces/tabs
523
524 for (size_t i = 0; i < input.length(); ++i) {
525 char current = input[i];
526
527 if (current == '\r' || current == '\t') {
528 current = ' ';
529 }
530
531 if (std::isspace(current)) {
532 if (current == '\n') {
533 // Add newline and reset inSpace
534 result += '\n';
535 inSpace = false;
536 } else if (!inSpace) {
537 // Only add space if it's the first whitespace in a sequence
538 if (!result.empty() && result.back() != '\n' && result.back() != ' ') {
539 result += ' ';
540 }
541 inSpace = true;
542 }
543 } else {
544 // Add non-space character and reset inSpace
545 result += current;
546 inSpace = false;
547 }
548 }
549
550 // Trim trailing space if any
551 if (!result.empty() && result.back() == ' ') {
552 result.pop_back();
553 }
554
555 return result;
556}
