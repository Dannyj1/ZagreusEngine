125void Engine::handleSetOptionCommand(const std::string& args) {
126 std::istringstream iss(args);
127 std::string arg;
128 std::string section;
129 std::string name;
130 std::string value;
131
132 while (iss >> arg) {
133 std::string lowercaseWord = arg;
134 std::ranges::transform(lowercaseWord, lowercaseWord.begin(),
135 [](const unsigned char c) { return std::tolower(c); });
136
137 if (lowercaseWord == "name") {
138 section = arg;
139 continue;
140 }
141
142 if (lowercaseWord == "value") {
143 section = arg;
144 continue;
145 }
146
147 if (section == "name") {
148 if (name.empty()) {
149 name = arg;
150 } else {
151 name += " " + arg;
152 }
153 }
154
155 if (section == "value") {
156 if (value.empty()) {
157 value = arg;
158 } else {
159 value += " " + arg;
160 }
161 }
162 }
163
164 if (name.empty()) {
165 sendMessage("ERROR: No option name provided.");
166 return;
167 }
168
169 if (!hasOption(name)) {
170 sendMessage("ERROR: Option " + name + " does not exist.");
171 return;
172 }
173
174 UCIOption& option = this->getOption(name);
175
176 if (value.empty()) {
177 if (option.getOptionType() == Button) {
178 if (option.getValue() == "true") {
179 value = "false";
180 } else {
181 value = "true";
182 }
183 } else {
184 sendMessage("ERROR: No option value provided.");
185 return;
186 }
187 }
188
189 option.setValue(value);
190
191 if (!didSetup) {
192 doSetup();
193 } else if (name == "Hash") {
194 TranspositionTable::getTT()->setTableSize(std::stoi(value));
195 }
196}

203void Engine::handlePositionCommand(const std::string_view args) {
204 // If the first arg is not "startpos" or "fen", report invalid usage
205 if (!args.starts_with("startpos") && !args.starts_with("fen")) {
206 sendMessage("ERROR: Invalid usage of position command.");
207 return;
208 }
209
210 // Go through each argument one by one
211 std::istringstream iss((args.data()));
212 std::string arg;
213
214 // If the first argument is "startpos", set the board to the starting position
215 iss >> arg;
216
217 if (arg == "startpos") {
218 board.setFromFEN(startPosFEN);
219 } else if (arg == "fen") {
220 std::string fen;
221
222 // Append args to the fen string until we reach the "moves" keyword or the end of the string
223 while (iss >> arg && arg != "moves") {
224 if (fen.empty()) {
225 fen = arg;
226 } else {
227 fen += " " + arg;
228 }
229 }
230
231 if (fen.empty()) {
232 sendMessage("ERROR: No FEN string provided.");
233 return;
234 }
235
236 if (!board.setFromFEN(fen)) {
237 sendMessage("ERROR: Invalid FEN string provided.");
238 return;
239 }
240 }
241
242 while (iss >> arg) {
243 if (arg == "moves") {
244 continue;
245 }
246
247 if (arg.size() != 4 && arg.size() != 5) {
248 sendMessage("ERROR: Invalid move notation provided.");
249 return;
250 }
251
252 Move move = getMoveFromMoveNotation(arg);
253
254 // TODO: Move this to the getMoveFromMoveNotation function, need to work around circular dependency first
255 Piece movedPiece = board.getPieceOnSquare(getFromSquare(move));
256
257 if (movedPiece == WHITE_KING) {
258 if (getFromSquare(move) == E1 && getToSquare(move) == G1 && board.canCastle<WHITE_KINGSIDE>()) {
259 move = encodeMove(E1, G1, CASTLING);
260 } else if (getFromSquare(move) == E1 && getToSquare(move) == C1 && board.canCastle<WHITE_QUEENSIDE>()) {
261 move = encodeMove(E1, C1, CASTLING);
262 }
263 } else if (movedPiece == BLACK_KING) {
264 if (getFromSquare(move) == E8 && getToSquare(move) == G8 && board.canCastle<BLACK_KINGSIDE>()) {
265 move = encodeMove(E8, G8, CASTLING);
266 } else if (getFromSquare(move) == E8 && getToSquare(move) == C8 && board.canCastle<BLACK_QUEENSIDE>()) {
267 move = encodeMove(E8, C8, CASTLING);
268 }
269 } else if (movedPiece == WHITE_PAWN || movedPiece == BLACK_PAWN) {
270 if (getToSquare(move) == board.getEnPassantSquare()) {
271 move = encodeMove(getFromSquare(move), getToSquare(move), EN_PASSANT);
272 }
273 }
274
275 board.makeMove(move);
276 }
277}

279void Engine::handleGoCommand(std::string_view args) {
280 if (!didSetup) {
281 doSetup();
282 }
283
284 // If the board is empty, set it to the starting position
285 if (board.getOccupiedBitboard() == 0ULL) {
286 if (!board.setFromFEN(startPosFEN)) {
287 sendMessage("ERROR: Could not initialize default position.");
288 return;
289 }
290 }
291
292 std::istringstream iss(args.data());
293 std::string arg;
294
295 // Parse arguments. Currently, only wtime and btime or supported, just ignore the others.
296 uint32_t whiteTime = 0;
297 uint32_t blackTime = 0;
298 uint32_t whiteInc = 0;
299 uint32_t blackInc = 0;
300 uint16_t depth = 0;
301
302 while (iss >> arg) {
303 if (arg == "wtime") {
304 iss >> whiteTime;
305 } else if (arg == "btime") {
306 iss >> blackTime;
307 } else if (arg == "depth") {
308 iss >> depth;
309 } else if (arg == "winc") {
310 iss >> whiteInc;
311 } else if (arg == "binc") {
312 iss >> blackInc;
313 }
314 }
315
316 if (whiteTime == 0 && blackTime == 0 && depth == 0) {
317 // TODO: implement support for other args
318 sendMessage("ERROR: No time control or depth limit provided.");
319 return;
320 }
321
322 this->searchStopped = false;
323 SearchParams params{};
324 SearchStats stats{};
325
326 params.whiteTime = whiteTime;
327 params.blackTime = blackTime;
328 params.whiteInc = whiteInc;
329 params.blackInc = blackInc;
330 params.depth = depth;
331
332 Move bestMove;
333
334 if (board.getSideToMove() == WHITE) {
335 bestMove = search<WHITE>(*this, board, params, stats);
336 } else {
337 bestMove = search<BLACK>(*this, board, params, stats);
338 }
339
340 sendMessage("bestmove " + getMoveNotation(bestMove));
341}

354void Engine::handlePerftCommand(const std::string& args) {
355 if (!didSetup) {
356 doSetup();
357 }
358
359 if (args.empty() || args == " " || args == "\n") {
360 sendMessage("ERROR: No depth provided.");
361 return;
362 }
363
364 if (args.contains(' ')) {
365 sendMessage("ERROR: Too many arguments provided.");
366 return;
367 }
368
369 int depth = 0;
370
371 try {
372 depth = std::stoi(args);
373 } catch (const std::invalid_argument& e) {
374 sendMessage("ERROR: Depth must be an integer.");
375 return;
376 }
377
378 if (depth <= 0) {
379 sendMessage("ERROR: Depth must be at least 1.");
380 return;
381 }
382
383 if (board.getOccupiedBitboard() == 0ULL) {
384 board.setFromFEN(startPosFEN);
385 }
386
387 const auto start = std::chrono::high_resolution_clock::now();
388 const uint64_t nodes = perft(board, depth);
389 const auto end = std::chrono::high_resolution_clock::now();
390 const std::string tookSeconds = std::to_string(std::chrono::duration<double>(end - start).count());
391
392 sendInfoMessage(
393 "Depth: " + std::to_string(depth) + ", Nodes: " + std::to_string(nodes) + ", Time: " + tookSeconds + "s");
394}

400void Engine::processCommand(const std::string_view command, const std::string& args) {
401 if (command == "uci") {
402 handleUciCommand();
403 } else if (command == "debug") {
404 handleDebugCommand(args);
405 } else if (command == "isready") {
406 handleIsReadyCommand(args);
407 } else if (command == "setoption") {
408 handleSetOptionCommand(args);
409 } else if (command == "register") {
410 // Ignore this command, not relevant to this engine
411 } else if (command == "ucinewgame") {
412 handleUciNewGameCommand();
413 } else if (command == "position") {
414 handlePositionCommand(args);
415 } else if (command == "go") {
416 std::thread searchThread{&Engine::handleGoCommand, this, args};
417
418 searchThread.detach();
419 } else if (command == "stop") {
420 handleStopCommand();
421 } else if (command == "ponderhit") {
422 handlePonderHitCommand(args);
423 } else if (command == "quit") {
424 handleQuitCommand(args);
425 } else if (command == "perft") {
426 handlePerftCommand(args);
427 } else if (command == "print") {
428 handlePrintCommand();
429 } else {
430 // If unknown, we must skip it and process the rest.
431 if (args.empty() || args == " " || args == "\n") {
432 std::cout << "Unknown command: " << command << std::endl;
433 return;
434 }
435
436 std::string newCommand;
437 std::string newArgs;
438
439 if (args.contains(' ')) {
440 newCommand = args.substr(0, args.find(' '));
441 newArgs = args.substr(args.find(' ') + 1);
442 } else {
443 newCommand = args;
444 }
445
446 processCommand(newCommand, newArgs);
447 }
448}

519std::string removeRedundantSpaces(const std::string_view input) {
520 std::string result;
521 bool inSpace = false; // Track if we are in a sequence of spaces/tabs
522
523 for (size_t i = 0; i < input.length(); ++i) {
524 char current = input[i];
525
526 if (current == '\r' || current == '\t') {
527 current = ' ';
528 }
529
530 if (std::isspace(current)) {
531 if (current == '\n') {
532 // Add newline and reset inSpace
533 result += '\n';
534 inSpace = false;
535 } else if (!inSpace) {
536 // Only add space if it's the first whitespace in a sequence
537 if (!result.empty() && result.back() != '\n' && result.back() != ' ') {
538 result += ' ';
539 }
540 inSpace = true;
541 }
542 } else {
543 // Add non-space character and reset inSpace
544 result += current;
545 inSpace = false;
546 }
547 }
548
549 // Trim trailing space if any
550 if (!result.empty() && result.back() == ' ') {
551 result.pop_back();
552 }
553
554 return result;
555}
