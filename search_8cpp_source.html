40void initializeSearch() {
41 for (int depth = 1; depth < MAX_PLIES; ++depth) {
42 for (int moveCount = 1; moveCount < MAX_MOVES; ++moveCount) {
43 lmrTable[depth][moveCount] = static_cast<int>(0.5 + 0.5 * std::log(depth) * std::log(moveCount));
44 }
45 }
46}

49// TODO: Support more search variables (infinite, max nodes, etc.)
50template <PieceColor color>
51Move search(Engine& engine, Board& board, SearchParams& params, SearchStats& stats) {
52 int depth = 1;
53 const int currentPly = board.getPly();
54 int searchTime = calculateSearchTime<color>(params);
55 const auto endTime = std::chrono::steady_clock::now() + std::chrono::milliseconds(searchTime);
56 const auto startTime = std::chrono::steady_clock::now();
57 PvLine bestPvLine = PvLine{board.getPly()};
58
59 engine.setSearchStopped(false);
60
61 while (!engine.isSearchStopped() && (currentPly + depth) < MAX_PLIES) {
62 if (params.blackTime > 0 || params.whiteTime > 0) {
63 // Don't start the next iteration if we are 10% away from the end time
64 if (std::chrono::steady_clock::now() + std::chrono::milliseconds(searchTime / 10) > endTime) {
65 engine.setSearchStopped(true);
66 break;
67 }
68 }
69
70 if (params.depth > 0 && depth > params.depth) {
71 engine.setSearchStopped(true);
72 break;
73 }
74
75 PvLine pvLine = PvLine{board.getPly()};
76
77 const int score = pvSearch<color, ROOT>(engine, board, INITIAL_ALPHA, INITIAL_BETA, depth, stats, endTime,
78 pvLine);
79 assert(score != INITIAL_ALPHA && score != INITIAL_BETA);
80 assert(depth > 0);
81
82 if (std::chrono::steady_clock::now() > endTime) {
83 engine.setSearchStopped(true);
84 break;
85 }
86
87 if (engine.isSearchStopped()) {
88 break;
89 }
90
91 bestPvLine = pvLine;
92 board.setPreviousPvLine(bestPvLine);
93
94 stats.score = score;
95 stats.timeSpentMs = std::chrono::duration_cast<std::chrono::milliseconds>(
96 std::chrono::steady_clock::now() - startTime).count();
97 stats.depth = depth;
98
99 // Make sure we don't divide by zero
100 if (stats.timeSpentMs == 0) {
101 stats.timeSpentMs = 1;
102 }
103
104 const uint64_t totalNodesSearch = stats.nodesSearched + stats.qNodesSearched;
105 const uint64_t nps = static_cast<double>(totalNodesSearch) / (
106 static_cast<double>(stats.timeSpentMs) / 1000.0);
107 std::string pvString = parsePvLine(bestPvLine);
108 engine.sendInfoMessage("depth " + std::to_string(stats.depth) + " score cp " + std::to_string(stats.score) +
109 " nodes " + std::to_string(totalNodesSearch) + " time " +
110 std::to_string(stats.timeSpentMs) + " nps " + std::to_string(nps) + " pv " + pvString);
111 depth += 1;
112 }
113
114 if (bestPvLine.moves[0] == NO_MOVE) {
115 // Find the first legal move and play that
116 MoveList moves = MoveList{};
117 generateMoves<color, ALL>(board, moves);
118 MovePicker movePicker{moves};
119 Move move;
120 Move bestMove = NO_MOVE;
121
122 while (movePicker.next(move)) {
123 board.makeMove(move);
124
125 if (!board.isPositionLegal<color>()) {
126 board.unmakeMove();
127 continue;
128 }
129
130 bestMove = move;
131 board.unmakeMove();
132 break;
133 }
134
135 assert(bestMove != NO_MOVE);
136 return bestMove;
137 }
138
139 assert(bestPvLine.moves[0] != NO_MOVE);
140 board.setPreviousPvLine(bestPvLine);
141 return bestPvLine.moves[0];
142}

147template <PieceColor color, NodeType nodeType>
148int pvSearch(Engine& engine, Board& board, int alpha, int beta, int depth, SearchStats& stats,
149 const std::chrono::time_point<std::chrono::steady_clock>& endTime, PvLine& pvLine) {
150 constexpr bool isPV = nodeType == PV || nodeType == ROOT;
151 constexpr bool isRoot = nodeType == ROOT;
152 constexpr PieceColor opponentColor = !color;
153
154 if (!isRoot && (stats.nodesSearched + stats.qNodesSearched) % 4096 == 0 && std::chrono::steady_clock::now() >
155 endTime) {
156 engine.setSearchStopped(true);
157 return beta;
158 }
159
160 bool isInCheck = board.isKingInCheck<color>();
161
162 if (isInCheck) {
163 depth += 1;
164 }
165
166 if (depth <= 0) {
167 assert(!isRoot);
168 pvLine.moveCount = 0;
169 return qSearch<color, nodeType>(engine, board, alpha, beta, depth, stats, endTime);
170 }
171
172 stats.nodesSearched += 1;
173
174 if (!isPV) {
175 // Check for a transposition table hit
176 const int16_t score = tt->probePosition(board.getZobristHash(), depth, alpha, beta, board.getPly());
177
178 if (score != NO_TT_SCORE) {
179 return score;
180 }
181
182 // Null Move Pruning
183 if (depth >= 3 && !isInCheck && board.hasNonPawnMaterial<color>() && !board.getPreviousMove() == NO_MOVE) {
184 board.makeNullMove();
185 const int R = 2 + depth / 3;
186 PvLine nmpPvLine = PvLine{board.getPly()};
187 const int nullMoveScore = -pvSearch<opponentColor, REGULAR>(engine, board, -beta, -beta + 1, depth - R,
188 stats, endTime, nmpPvLine);
189 board.unmakeNullMove();
190
191 if (nullMoveScore >= beta) {
192 return nullMoveScore;
193 }
194 }
195 }
196
197 bool firstMove = true;
198 int legalMoves = 0;
199
200 Move move;
201 MoveList moves = MoveList{};
202
203 if (isInCheck) {
204 generateMoves<color, EVASIONS>(board, moves);
205 } else {
206 generateMoves<color, ALL>(board, moves);
207 }
208
209 MoveList searchedQuietMoves{};
210 MovePicker movePicker{moves};
211 movePicker.score(board);
212 PvLine nodePvLine = PvLine{board.getPly()};
213 Move bestMove = NO_MOVE;
214 int bestScore = INT32_MIN;
215 int movesSearched = 0;
216
217 while (movePicker.next(move)) {
218 const Square toSquare = getToSquare(move);
219 const Piece capturedPiece = board.getPieceOnSquare(toSquare);
220
221 board.makeMove(move);
222
223 if (!board.isPositionLegal<color>()) {
224 board.unmakeMove();
225 continue;
226 }
227
228 legalMoves += 1;
229
230 if (capturedPiece == EMPTY) {
231 searchedQuietMoves.moves[searchedQuietMoves.size++] = move;
232 }
233
234 bool doFullSearch = true;
235 int score = INT32_MIN;
236
237
238 // Late Move Reduction
239 if (movesSearched > 1 && depth >= 3 && !(isPV && capturedPiece != EMPTY) && !(
240 isPV && getMoveType(move) == PROMOTION)) {
241 doFullSearch = false;
242 int R = 0;
243 const Square opponentKingSquare = bitboardToSquare(
244 board.getPieceBoard<color == WHITE ? BLACK_KING : WHITE_KING>());
245 const uint64_t opponentKingAttackers = board.getSquareAttackersByColor<color>(opponentKingSquare);
246
247 R = lmrTable[depth][movesSearched];
248 R -= isPV;
249 R -= isInCheck;
250 R -= opponentKingAttackers != 0;
251
252 // Make sure depth - 1 - R is at least 1
253 if (depth - 1 - R <= 0) {
254 R = depth - 2;
255 }
256
257 R = std::max(0, R);
258
259 score = -pvSearch<opponentColor, REGULAR>(engine, board, -alpha - 1, -alpha, depth - 1 - R, stats,
260 endTime, nodePvLine);
261
262 if (score > alpha) {
263 doFullSearch = true;
264 }
265 }
266
267 if (doFullSearch) {
268 if (firstMove) {
269 if (isRoot) {
270 score = -pvSearch<opponentColor, PV>(engine, board, -beta, -alpha, depth - 1, stats, endTime,
271 nodePvLine);
272 } else {
273 score = -pvSearch<opponentColor, nodeType>(engine, board, -beta, -alpha, depth - 1, stats, endTime,
274 nodePvLine);
275 }
276
277 firstMove = false;
278 } else {
279 score = -pvSearch<opponentColor, REGULAR>(engine, board, -alpha - 1, -alpha, depth - 1, stats, endTime,
280 nodePvLine);
281
282 if (isPV && score > alpha) {
283 score = -pvSearch<opponentColor, PV>(engine, board, -beta, -alpha, depth - 1, stats, endTime,
284 nodePvLine);
285 }
286 }
287 }
288
289 board.unmakeMove();
290
291 if (score >= beta) {
292 if (!engine.isSearchStopped()) {
293 if (capturedPiece == EMPTY) {
294 const int historyValue = 300 * depth - 250;
295
296 tt->updateHistory<color>(move, historyValue);
297
298 for (int i = 0; i < searchedQuietMoves.size; ++i) {
299 const Move quietMove = searchedQuietMoves.moves[i];
300
301 if (quietMove != move) {
302 tt->updateHistory<color>(quietMove, -historyValue);
303 }
304 }
305 }
306
307 if (!isRoot) {
308 tt->savePosition(board.getZobristHash(), depth, board.getPly(), score, bestMove, BETA);
309 }
310 }
311
312 return score;
313 }
314
315 if (score > bestScore) {
316 bestScore = score;
317
318 if (score > alpha) {
319 bestMove = move;
320 alpha = score;
321 pvLine.moves[0] = move;
322 std::memcpy(pvLine.moves + 1, nodePvLine.moves, nodePvLine.moveCount * sizeof(Move));
323 pvLine.moveCount = nodePvLine.moveCount + 1;
324 }
325 }
326
327 movesSearched += 1;
328 }
329
330 if (!legalMoves) {
331 if (isInCheck) {
332 alpha = -MATE_SCORE + board.getPly();
333 bestScore = alpha;
334 } else {
335 alpha = DRAW_SCORE;
336 bestScore = DRAW_SCORE;
337 }
338 } else {
339 if (!isRoot && board.isDraw()) {
340 alpha = DRAW_SCORE;
341 bestScore = DRAW_SCORE;
342 }
343 }
344
345 if (!isRoot) {
346 TTNodeType ttNodeType = ALPHA;
347
348 if (isPV) {
349 ttNodeType = EXACT;
350 }
351
352 if (!engine.isSearchStopped()) {
353 tt->savePosition(board.getZobristHash(), depth, board.getPly(), bestScore, bestMove, ttNodeType);
354 }
355 }
356
357 assert(bestScore != INITIAL_ALPHA);
358 return bestScore;
359}

361template <PieceColor color, NodeType nodeType>
362int qSearch(Engine& engine, Board& board, int alpha, int beta, int depth, SearchStats& stats,
363 const std::chrono::time_point<std::chrono::steady_clock>& endTime) {
364 assert(nodeType != ROOT);
365 constexpr bool isPV = nodeType == PV;
366
367 if ((stats.nodesSearched + stats.qNodesSearched) % 4096 == 0 && std::chrono::steady_clock::now() > endTime) {
368 engine.setSearchStopped(true);
369 return beta;
370 }
371
372 if (!isPV) {
373 const int16_t score = tt->probePosition(board.getZobristHash(), depth, alpha, beta, board.getPly());
374
375 if (score != NO_TT_SCORE) {
376 return score;
377 }
378 }
379
380 stats.qNodesSearched += 1;
381
382 const bool isInCheck = board.isKingInCheck<color>();
383 int bestScore = Evaluation(board).evaluate();
384
385 if (!isInCheck) {
386 // Stand pat
387 if (bestScore >= beta) {
388 if (!engine.isSearchStopped()) {
389 tt->savePosition(board.getZobristHash(), depth, board.getPly(), bestScore, NO_MOVE, BETA);
390 }
391
392 return bestScore;
393 }
394
395 if (bestScore > alpha) {
396 alpha = bestScore;
397 }
398 }
399
400 int legalMoves = 0;
401 Move move;
402 MoveList moves = MoveList{};
403 Move bestMove = NO_MOVE;
404
405 if (isInCheck) {
406 generateMoves<color, EVASIONS>(board, moves);
407 } else {
408 generateMoves<color, QSEARCH>(board, moves);
409 }
410
411 MovePicker movePicker{moves};
412 movePicker.score(board);
413
414 while (movePicker.next(move)) {
415 const Square toSquare = getToSquare(move);
416 const Piece capturedPiece = board.getPieceOnSquare(toSquare);
417
418 if (!isInCheck && capturedPiece != EMPTY) {
419 const bool see = board.see(move, 0);
420
421 if (!see) {
422 continue;
423 }
424 }
425
426 board.makeMove(move);
427
428 if (!board.isPositionLegal<color>()) {
429 board.unmakeMove();
430 continue;
431 }
432
433 legalMoves += 1;
434
435 const int score = -qSearch<!color, nodeType>(engine, board, -beta, -alpha, depth - 1, stats, endTime);
436
437 board.unmakeMove();
438
439 if (score >= beta) {
440 if (!engine.isSearchStopped()) {
441 tt->savePosition(board.getZobristHash(), depth, board.getPly(), score, bestMove, BETA);
442 }
443
444 return score;
445 }
446
447 if (score > bestScore) {
448 bestScore = score;
449
450 if (score > alpha) {
451 bestMove = move;
452 alpha = score;
453 }
454 }
455 }
456
457 if (!legalMoves && isInCheck) {
458 bestScore = -MATE_SCORE + board.getPly();
459 } else {
460 if (board.isDraw()) {
461 bestScore = DRAW_SCORE;
462 }
463 }
464
465 TTNodeType ttNodeType = ALPHA;
466
467 if (isPV) {
468 ttNodeType = EXACT;
469 }
470
471 if (!engine.isSearchStopped()) {
472 tt->savePosition(board.getZobristHash(), depth, board.getPly(), bestScore, bestMove, ttNodeType);
473 }
474
475 assert(bestScore != INITIAL_ALPHA);
476 return bestScore;
477}
