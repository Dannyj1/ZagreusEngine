43// TODO: Support more search variables (infinite, max nodes, etc.)
44template <PieceColor color>
45Move search(Engine& engine, Board& board, SearchParams& params, SearchStats& stats) {
46 int depth = 1;
47 const int currentPly = board.getPly();
48 int searchTime = calculateSearchTime<color>(params);
49 const auto endTime = std::chrono::steady_clock::now() + std::chrono::milliseconds(searchTime);
50 const auto startTime = std::chrono::steady_clock::now();
51 PvLine bestPvLine = PvLine{board.getPly()};
52
53 engine.setSearchStopped(false);
54
55 while (!engine.isSearchStopped() && (currentPly + depth) < MAX_PLY) {
56 if (params.blackTime > 0 || params.whiteTime > 0) {
57 // Don't start the next iteration if we are 10% away from the end time
58 if (std::chrono::steady_clock::now() + std::chrono::milliseconds(searchTime / 10) > endTime) {
59 engine.setSearchStopped(true);
60 break;
61 }
62 }
63
64 if (params.depth > 0 && depth > params.depth) {
65 engine.setSearchStopped(true);
66 break;
67 }
68
69 PvLine pvLine = PvLine{board.getPly()};
70
71 const int score = pvSearch<color, ROOT>(engine, board, INITIAL_ALPHA, INITIAL_BETA, depth, stats, endTime,
72 pvLine);
73 assert(score != INITIAL_ALPHA && score != INITIAL_BETA);
74 assert(depth > 0);
75
76 if (std::chrono::steady_clock::now() > endTime) {
77 engine.setSearchStopped(true);
78 break;
79 }
80
81 if (engine.isSearchStopped()) {
82 break;
83 }
84
85 bestPvLine = pvLine;
86 board.setPreviousPvLine(bestPvLine);
87
88 stats.score = score;
89 stats.timeSpentMs = std::chrono::duration_cast<std::chrono::milliseconds>(
90 std::chrono::steady_clock::now() - startTime).count();
91 stats.depth = depth;
92
93 // Make sure we don't divide by zero
94 if (stats.timeSpentMs == 0) {
95 stats.timeSpentMs = 1;
96 }
97
98 const uint64_t totalNodesSearch = stats.nodesSearched + stats.qNodesSearched;
99 const uint64_t nps = static_cast<double>(totalNodesSearch) / (
100 static_cast<double>(stats.timeSpentMs) / 1000.0);
101 std::string pvString = parsePvLine(bestPvLine);
102 engine.sendInfoMessage("depth " + std::to_string(stats.depth) + " score cp " + std::to_string(stats.score) +
103 " nodes " + std::to_string(totalNodesSearch) + " time " +
104 std::to_string(stats.timeSpentMs) + " nps " + std::to_string(nps) + " pv " + pvString);
105 depth += 1;
106 }
107
108 if (bestPvLine.moves[0] == NO_MOVE) {
109 // Find the first legal move and play that
110 MoveList moves = MoveList{};
111 generateMoves<color, ALL>(board, moves);
112 MovePicker movePicker{moves};
113 Move move;
114 Move bestMove = NO_MOVE;
115
116 while (movePicker.next(move)) {
117 board.makeMove(move);
118
119 if (!board.isPositionLegal<color>()) {
120 board.unmakeMove();
121 continue;
122 }
123
124 bestMove = move;
125 board.unmakeMove();
126 break;
127 }
128
129 assert(bestMove != NO_MOVE);
130 return bestMove;
131 }
132
133 assert(bestPvLine.moves[0] != NO_MOVE);
134 board.setPreviousPvLine(bestPvLine);
135 return bestPvLine.moves[0];
136}

141template <PieceColor color, NodeType nodeType>
142int pvSearch(Engine& engine, Board& board, int alpha, int beta, int depth, SearchStats& stats,
143 const std::chrono::time_point<std::chrono::steady_clock>& endTime, PvLine& pvLine) {
144 constexpr bool isPV = nodeType == PV || nodeType == ROOT;
145 constexpr bool isRoot = nodeType == ROOT;
146 constexpr PieceColor opponentColor = !color;
147
148 if (!isRoot && (stats.nodesSearched + stats.qNodesSearched) % 4096 == 0 && std::chrono::steady_clock::now() >
149 endTime) {
150 engine.setSearchStopped(true);
151 return beta;
152 }
153
154 if (depth <= 0) {
155 assert(!isRoot);
156 pvLine.moveCount = 0;
157 return qSearch<color, nodeType>(engine, board, alpha, beta, depth, stats, endTime);
158 }
159
160 stats.nodesSearched += 1;
161 // Move ttMove = NO_MOVE;
162
163 if (!isPV) {
164 const int16_t score = tt->probePosition(board.getZobristHash(), depth, alpha, beta, board.getPly());
165
166 if (score != NO_TT_SCORE) {
167 return score;
168 }
169 } /* else {
170 const TTEntry* ttEntry = tt->getEntry(board.getZobristHash());
171
172 if (ttEntry) {
173 ttMove = ttEntry->bestMove;
174 }
175 }*/
176
177 /*if (isPV && depth >= 5 && ttMove == NO_MOVE) {
178 depth -= 1;
179 }*/
180
181 bool firstMove = true;
182 int legalMoves = 0;
183
184 Move move;
185 MoveList moves = MoveList{};
186 bool isInCheck = board.isKingInCheck<color>();
187
188 if (isInCheck) {
189 generateMoves<color, EVASIONS>(board, moves);
190 } else {
191 generateMoves<color, ALL>(board, moves);
192 }
193
194 MoveList searchedQuietMoves{};
195 MovePicker movePicker{moves};
196 movePicker.sort(board);
197 Move bestMove = NO_MOVE;
198 PvLine nodePvLine = PvLine{board.getPly()};
199
200 while (movePicker.next(move)) {
201 const Square toSquare = getToSquare(move);
202 const Piece capturedPiece = board.getPieceOnSquare(toSquare);
203
204 board.makeMove(move);
205
206 if (!board.isPositionLegal<color>()) {
207 board.unmakeMove();
208 continue;
209 }
210
211 legalMoves += 1;
212
213 if (capturedPiece == EMPTY) {
214 searchedQuietMoves.moves[searchedQuietMoves.size++] = move;
215 }
216
217 int score;
218
219 if (firstMove) {
220 if (isRoot) {
221 score = -pvSearch<opponentColor, PV>(engine, board, -beta, -alpha, depth - 1, stats, endTime,
222 nodePvLine);
223 } else {
224 score = -pvSearch<opponentColor, nodeType>(engine, board, -beta, -alpha, depth - 1, stats, endTime,
225 nodePvLine);
226 }
227
228 firstMove = false;
229 } else {
230 score = -pvSearch<opponentColor, REGULAR>(engine, board, -alpha - 1, -alpha, depth - 1, stats, endTime,
231 nodePvLine);
232
233 if (isPV && score > alpha) {
234 score = -pvSearch<opponentColor, PV>(engine, board, -beta, -alpha, depth - 1, stats, endTime,
235 nodePvLine);
236 }
237 }
238
239 board.unmakeMove();
240
241 if (score >= beta) {
242 if (!engine.isSearchStopped()) {
243 bestMove = move;
244
245 if (capturedPiece == EMPTY) {
246 const int historyValue = 300 * depth - 250;
247
248 tt->updateHistory<color>(move, historyValue);
249
250 for (int i = 0; i < searchedQuietMoves.size; ++i) {
251 const Move quietMove = searchedQuietMoves.moves[i];
252
253 if (quietMove != move) {
254 tt->updateHistory<color>(quietMove, -historyValue);
255 }
256 }
257 }
258
259 tt->savePosition(board.getZobristHash(), depth, board.getPly(), score, bestMove, BETA);
260 }
261
262 return score;
263 }
264
265 if (score > alpha) {
266 alpha = score;
267 // bestMove = move;
268 pvLine.moves[0] = move;
269 std::memcpy(pvLine.moves + 1, nodePvLine.moves, nodePvLine.moveCount * sizeof(Move));
270 pvLine.moveCount = nodePvLine.moveCount + 1;
271 }
272 }
273
274 if (!legalMoves) {
275 if (isInCheck) {
276 alpha = -MATE_SCORE + board.getPly();
277 } else {
278 alpha = DRAW_SCORE;
279 }
280 } else {
281 if (!isRoot && board.isDraw()) {
282 return DRAW_SCORE;
283 }
284 }
285
286 if (!isRoot) {
287 TTNodeType ttNodeType = ALPHA;
288
289 if (isPV) {
290 ttNodeType = EXACT;
291 }
292
293 if (!engine.isSearchStopped()) {
294 tt->savePosition(board.getZobristHash(), depth, board.getPly(), alpha, bestMove, ttNodeType);
295 }
296 }
297
298 assert(alpha != INITIAL_ALPHA);
299 return alpha;
300}

302template <PieceColor color, NodeType nodeType>
303int qSearch(Engine& engine, Board& board, int alpha, int beta, int depth, SearchStats& stats,
304 const std::chrono::time_point<std::chrono::steady_clock>& endTime) {
305 assert(nodeType != ROOT);
306 constexpr bool isPV = nodeType == PV;
307
308 if (!isPV) {
309 const int16_t score = tt->probePosition(board.getZobristHash(), depth, alpha, beta, board.getPly());
310
311 if (score != NO_TT_SCORE) {
312 return score;
313 }
314 }
315
316 if ((stats.nodesSearched + stats.qNodesSearched) % 4096 == 0 && std::chrono::steady_clock::now() > endTime) {
317 engine.setSearchStopped(true);
318 return beta;
319 }
320
321 stats.qNodesSearched += 1;
322
323 int bestScore = Evaluation(board).evaluate();
324
325 if (bestScore >= beta) {
326 if (!engine.isSearchStopped()) {
327 tt->savePosition(board.getZobristHash(), depth, board.getPly(), bestScore, NO_MOVE, BETA);
328 }
329
330 return bestScore;
331 }
332
333 if (bestScore > alpha) {
334 alpha = bestScore;
335 }
336
337 const bool isInCheck = board.isKingInCheck<color>();
338 int legalMoves = 0;
339 Move move;
340 MoveList moves = MoveList{};
341
342 if (isInCheck) {
343 generateMoves<color, EVASIONS>(board, moves);
344 } else {
345 generateMoves<color, QSEARCH>(board, moves);
346 }
347
348 MovePicker movePicker{moves};
349 movePicker.sort(board);
350
351 while (movePicker.next(move)) {
352 const Square toSquare = getToSquare(move);
353 const Piece capturedPiece = board.getPieceOnSquare(toSquare);
354
355 if (!isInCheck && capturedPiece != EMPTY) {
356 const bool see = board.see(move, 0);
357
358 if (!see) {
359 continue;
360 }
361 }
362
363 board.makeMove(move);
364
365 if (!board.isPositionLegal<color>()) {
366 board.unmakeMove();
367 continue;
368 }
369
370 legalMoves += 1;
371
372 const int score = -qSearch<!color, nodeType>(engine, board, -beta, -alpha, depth - 1, stats, endTime);
373
374 board.unmakeMove();
375
376 if (score >= beta) {
377 if (!engine.isSearchStopped()) {
378 tt->savePosition(board.getZobristHash(), depth, board.getPly(), score, NO_MOVE, BETA);
379 }
380
381 return score;
382 }
383
384 if (score > bestScore) {
385 bestScore = score;
386 }
387
388 if (score > alpha) {
389 alpha = score;
390 }
391 }
392
393 if (!legalMoves && isInCheck) {
394 bestScore = -MATE_SCORE + board.getPly();
395 } else {
396 if (board.isDraw()) {
397 return DRAW_SCORE;
398 }
399 }
400
401 TTNodeType ttNodeType = ALPHA;
402
403 if (isPV) {
404 ttNodeType = EXACT;
405 }
406
407 if (!engine.isSearchStopped()) {
408 tt->savePosition(board.getZobristHash(), depth, board.getPly(), bestScore, NO_MOVE, ttNodeType);
409 }
410
411 assert(alpha != INITIAL_ALPHA);
412 return bestScore;
413}
